let wallpapers = [ { url: "https://images.unsplash.com/photo-1500530855697-b586d89ba3ee", category: "nature" }, { url: "https://images.unsplash.com/photo-1519681393784-d120267933ba", category: "nature" }, { url: "https://images.unsplash.com/photo-1518770660439-4636190af475", category: "cars" }, { url: "https://images.unsplash.com/photo-1549924231-f129b911e442", category: "anime" } ]; /* RENDER */ function renderWallpapers(category) { let gallery = document.getElementById("gallery"); gallery.innerHTML = ""; wallpapers.forEach(w => { if (category === "all" || w.category === category) { let div = document.createElement("div"); div.className = "wallpaper"; div.innerHTML = ` `; gallery.appendChild(div); } }); } /* FILTER */ function filter(category) { renderWallpapers(category); } /* SEARCH */ function searchWallpapers() { let input = document.getElementById("searchBox").value.toLowerCase(); let filtered = wallpapers.filter(w => w.category.toLowerCase().includes(input) ); renderFiltered(filtered); } /* FILTERED RENDER */ function renderFiltered(list) { let gallery = document.getElementById("gallery"); gallery.innerHTML = ""; list.forEach(w => { let div = document.createElement("div"); div.className = "wallpaper"; div.innerHTML = ` `; gallery.appendChild(div); }); } /* VIEWER */ function openViewer(url) { document.getElementById("viewerImg").src = url; document.getElementById("downloadBtn").href = url; document.getElementById("viewer").style.display = "flex"; } function closeViewer() { document.getElementById("viewer").style.display = "none"; } /* INITIAL LOAD */ renderWallpapers("all");
